    ARRIBA, ABAJO, IZQUIERDA, DERECHA

    IGUAL (=), MENOR (<), MAYOR (>), MENOR_IGUAL (<=), MAYOR_IGUAL (>=)
    DIFERENTE (<>), AND (&&), OR (||), NOT (!)

Delimitadores:
    COMA (,), PUNTO_COMA (;), PUNTO (.)
    PAREN_IZQ '(', PAREN_DER ')'
    CORCHETE_IZQ '[', CORCHETE_DER ']'

Literales:
    IDENTIFICADOR, NUMERO, CADENA

Especiales:
    EOF (fin de archivo)
